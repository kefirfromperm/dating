022    package pl.burningice.plugins.image.ast;
023    
024    import org.codehaus.groovy.ast.*;
025    import org.codehaus.groovy.transform.*;
026    import org.codehaus.groovy.control.*;
027    import java.lang.reflect.Modifier;
028    import pl.burningice.plugins.image.ast.intarface.FileImageContainer;
029    
030    /**
031     * Object execute transformation of object marked by FileImageContainer annotation
032     *
033     * @author pawel.gdula@burningice.pl
034     */
035    @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
036    public class FileImageContainerTransformation extends AbstractImageContainerTransformation {
037    
038        @Override
039        protected void transformSpecified(ClassNode node, String fieldName) {
040            // implements interface
041            node.addInterface(new ClassNode(FileImageContainer.class));
042            // imageExtension field
043            FieldNode imageExtension = new FieldNode("imageExtension", Modifier.PRIVATE, new ClassNode(String.class), new ClassNode(node.getClass()), null);
044            node.addField(imageExtension);
045            addGetter(imageExtension, node);
046            addSetter(imageExtension, node);
047            addNullableConstraint(node, "imageExtension");
048        }
049    }
