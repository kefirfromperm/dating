022    package pl.burningice.plugins.image.ast;
023    
024    import org.codehaus.groovy.transform.GroovyASTTransformationClass;
025    import java.lang.annotation.*;
026    
027    /**
028     * Annotation that allows to mark domain object that will act as image container
029     * for files saved in specified dir (non db storage)
030     *
031     * @author pawel.gdula@burningice.pl
032     */
033    @Retention(RetentionPolicy.RUNTIME)
034    @Target({ElementType.TYPE})
035    @GroovyASTTransformationClass("pl.burningice.plugins.image.ast.FileImageContainerTransformation")
036    public @interface FileImageContainer {
037        String field ();
038    }
