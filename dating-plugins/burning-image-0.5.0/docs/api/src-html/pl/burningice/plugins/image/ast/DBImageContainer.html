022    package pl.burningice.plugins.image.ast;
023    
024    import org.codehaus.groovy.transform.GroovyASTTransformationClass;
025    
026    import java.lang.annotation.ElementType;
027    import java.lang.annotation.Retention;
028    import java.lang.annotation.RetentionPolicy;
029    import java.lang.annotation.Target;
030    
031    /**
032     * Annotation that allows to mark domain object that will act as image container
033     * for files saved in database
034     *
035     * @author pawel.gdula@burningice.pl
036     */
037    @Retention(RetentionPolicy.RUNTIME)
038    @Target({ElementType.TYPE})
039    @GroovyASTTransformationClass("pl.burningice.plugins.image.ast.DBImageContainerTransformation")
040    public @interface DBImageContainer {
041        String field ();
042    }
